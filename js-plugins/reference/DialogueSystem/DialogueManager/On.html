DialogueManager.On

void On (DialogueEvent event, function callback);

Parameters

event The event the callback will be called/removed from.
callback The callback which will be called/removed.

Description

Adds/Removes a callback for a specified event. 

JavaScriptconst dialogueBox = document.querySelector("div");

const manager = new DialogueManager(dialogueBox);

// Everytime the manager is loaded (once),
// execute the callback
manager.On(DialogueEvent.Load, async () => {
    await manager.SetActive(true);
    await manager.Type("When the impostor is sus.", 1);
    await DialogueLoop.Delay(1);
    await manager.SetActive(false);
});

// Everytime the system disables,
// execute the callback
manager.On(DialogueEvent.Disable, async () => {
    if (!confirm("à¶ž")) return;
    
    // Click "Ok" to run these
    await manager.SetActive(true);
    await manager.Type("AMOGUS", 1);
    await DialogueLoop.Delay(1);
    await manager.SetActive(false);
});
