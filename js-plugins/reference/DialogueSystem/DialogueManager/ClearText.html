DialogueManager.ClearText

async void ClearText ();

Description

Clears the dialogue line.

Use this if you only want to clear text. 

JavaScriptconst dialogueBox = document.querySelector("div");

const manager = new DialogueManager(dialogueBox);

manager.config.sprites.items = [
    new DialogueSprite("sus", "../img/amogus.png")
];

manager.Once(DialogueEvent.Load, async () => {
    await manager.SetActive(true);
    await manager.Type("The only non themed portrait I have here is", 1);
    await manager.Type("...", 0.02);
    
    // Clear text
    await manager.ClearText();
    
    // Use portrait
    await manager.Type("This horrendous thing.", 1, new DialogueData({
        portrait : "sus"
    }));
    await DialogueLoop.Delay(1);
    
    // This will not clear the portrait
    await manager.ClearText();
    
    await manager.Type("I don't know what else to make.", 1);
});
