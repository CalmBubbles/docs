DialogueManager.SetActive

async void SetActive (boolean state);

Parameters

state The state the manager will be changed to.

Description

Enables/Disables the manager.

You are required to use this method to enable most of the manager's methods. 

JavaScriptconst dialogueBox = document.querySelector("div");

const manager = new DialogueManager(dialogueBox);

manager.animation.box.onEnable = new DialogueAnimation({
    animation : "box",
    duration : 0.25
});

manager.Once(DialogueEvent.Load, async () => {
    // Not all tasks will run
    await manager.Type("I can't talk,", 0.25);
    await DialogueLoop.Delay(1);
    await manager.Type(" h-help...", 0.125);
    await DialogueLoop.Delay(5);
    
    // Enable the system
    await manager.SetActive(true);
    
    // All tasks will run
    await manager.Type("Oh hey,", 0.75);
    await DialogueLoop.Delay(0.5);
    await manager.Type(" I can talk now.", 1);
    await DialogueLoop.Delay(1.5);
    await manager.ClearText();
    await manager.Type("Uhhhhhhhhh...", 0.25);
    await DialogueLoop.Delay(1.25);
    await manager.ClearText();
    await manager.Type("Anyway,", 1);
    await DialogueLoop.Delay(0.25);
    await manager.Type(" I was going to tell you about-", 1.25);
    
    // Disable the system
    await manager.SetActive(false);
    
    // This task will not run
    await manager.Type("WHYYYYYYYYYYYY!??!", 0.125);
});
