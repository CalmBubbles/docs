DialogueLoop.Delay

static async void Delay (number time);

Parameters

time Amount of time in seconds the delay takes.

Description

Waits for a specific amount of time.

Use this within asynchronous methods to add delay to tasks. 
JavaScript// Adds text with line break
function Append (text)
{
    const br = document.createElement("br");
    
    document.body.append(text, br);
}

async function Run ()
{
    Append("Waiting for 5 seconds...");
    
    await DialogueLoop.Delay(5);
    
    Append("Done waiting");
    
    await DialogueLoop.Delay(2);
    
    Append("Counting to 10...");
    
    await DialogueLoop.Delay(0.5)
    
    // Count to 10
    for (let i = 1; i <= 10; i++) Append(i);
    
    await DialogueLoop.Delay(0.5)
    
    Append("Bruh, too fast");
    
    await DialogueLoop.Delay(2);
    
    Append("Counting to 10, but with delay...");
    
    await DialogueLoop.Delay(0.5)
    
    for (let i = 1; i <= 10; i++)
    {
        // Wait for a sec every number
        await DialogueLoop.Delay(1);
        
        Append(i);
    }
    
    await DialogueLoop.Delay(0.5)
    
    Append("Done counting");
}

Run();
