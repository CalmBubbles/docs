DialogueLoop.Append

static void Append (function callback);
static void Append (function callback, number delay);
static void Append (function callback, number delay, boolean function shouldClear);

Parameters

callback The task to be called.
delay The time in seconds it takes for the callback to be called.
shouldClear Specifies if the task should be disposed.

Description

Adds a method/task to be updated. 

HTML<!--Set text-->
loaded : <span id="loaded">false</span><br>
count : <span id="count">0</span><br>
seconds : <span id="seconds">0</span>
JavaScript// Get text
const lSpan = document.querySelector("#loaded");
const cSpan = document.querySelector("#count");
const sSpan = document.querySelector("#seconds");

let loaded = false;
let counter = 0;

DialogueLoop.Append(
    // Add Load as task
    () => Load(),
    
    // Wait for a second to call task
    1,
    
    // Clear task if loaded
    () => loaded
);

function Load ()
{
    loaded = true;
    
    lSpan.textContent = "true";
    
    // Add Update as task
    DialogueLoop.Append(() => Update());
    
    // Increase counter every 2.5 seconds
    DialogueLoop.Append(() => { counter++; }, 2.5);
}

// Update text
function Update ()
{
    cSpan.textContent = counter;
    sSpan.textContent = parseInt(DialogueLoop.time);
}
